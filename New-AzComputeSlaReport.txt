<#
.SYNOPSIS
    Generates a monthly Azure Compute Availability SLA report across an entire tenant.

.DESCRIPTION
    For every VM and VMSS instance the user has access to, this script:
      1. Determines the committed Microsoft SLA tier based on configuration:
            - Availability Zones (>=2 zones spread)         -> 99.99%
            - Availability Set                              -> 99.95%
            - Single instance, all Premium SSD / Ultra      -> 99.9%
            - Single instance, Standard SSD OS disk         -> 99.5%
            - Single instance, Standard HDD OS disk         -> 95.0%
      2. Queries Azure Resource Health (AvailabilityStatuses) for the
         reporting month and calculates measured availability %
         (time in 'Available' / total reporting seconds).
      3. Flags whether the measured availability met the committed SLA.
      4. Exports CSV + HTML summary to the output folder.

.PARAMETER Month
    Reporting month as 'yyyy-MM'. Defaults to previous month.

.PARAMETER OutputFolder
    Where to write the report files. Defaults to script folder \ Reports.

.PARAMETER SubscriptionId
    Optional. Limit to one subscription. If omitted, all accessible subs are used.

.EXAMPLE
    .\New-AzComputeSlaReport.ps1
    .\New-AzComputeSlaReport.ps1 -Month 2026-03
    .\New-AzComputeSlaReport.ps1 -SubscriptionId 00000000-0000-0000-0000-000000000000

.NOTES
    Requires modules: Az.Accounts, Az.ResourceGraph, Az.ResourceHealth (or REST fallback).
    Run: Connect-AzAccount   before executing.
#>

[CmdletBinding()]
param(
    [string]$Month = (Get-Date).AddMonths(-1).ToString('yyyy-MM'),
    [string]$OutputFolder = (Join-Path $PSScriptRoot 'Reports'),
    [string]$SubscriptionId,
    [int]$MatrixMonths = 12   # how many months back to include in the region/month matrix
)

#region Setup ------------------------------------------------------------------
$ErrorActionPreference = 'Stop'

function Ensure-Module {
    param([string]$Name)
    if (-not (Get-Module -ListAvailable -Name $Name)) {
        Write-Host "Installing module $Name ..." -ForegroundColor Yellow
        Install-Module $Name -Scope CurrentUser -Force -AllowClobber
    }
    Import-Module $Name -ErrorAction Stop
}

Ensure-Module Az.Accounts
Ensure-Module Az.ResourceGraph

if (-not (Get-AzContext)) {
    Write-Host 'Connecting to Azure...' -ForegroundColor Cyan
    Connect-AzAccount | Out-Null
}

if (-not (Test-Path $OutputFolder)) { New-Item -ItemType Directory -Path $OutputFolder | Out-Null }

# Reporting window
$reportStart = [datetime]::ParseExact("$Month-01", 'yyyy-MM-dd', $null).ToUniversalTime()
$reportEnd   = $reportStart.AddMonths(1)
$now         = [datetime]::UtcNow
if ($reportEnd -gt $now) { $reportEnd = $now }
$totalSeconds = ($reportEnd - $reportStart).TotalSeconds

Write-Host "Reporting window (UTC): $reportStart  ->  $reportEnd" -ForegroundColor Cyan
#endregion

#region Inventory via Resource Graph ------------------------------------------
$kqlVm = @'
Resources
| where type == "microsoft.compute/virtualmachines"
| extend zonesStr          = tostring(zones),
         availabilitySetId = tostring(properties.availabilitySet.id),
         vmssId            = tostring(properties.virtualMachineScaleSet.id),
         osDiskType        = tostring(properties.storageProfile.osDisk.managedDisk.storageAccountType),
         vmSize             = tostring(properties.hardwareProfile.vmSize),
         resKind            = "VM"
| project subscriptionId, resourceGroup, name, id, location,
          zonesStr, availabilitySetId, vmssId, osDiskType, vmSize, resKind
'@

$kqlVmss = @'
Resources
| where type == "microsoft.compute/virtualmachinescalesets"
| extend zonesStr   = tostring(zones),
         osDiskType = tostring(properties.virtualMachineProfile.storageProfile.osDisk.managedDisk.storageAccountType),
         vmSize     = tostring(sku.name),
         availabilitySetId = "",
         vmssId            = "",
         resKind           = "VMSS"
| project subscriptionId, resourceGroup, name, id, location,
          zonesStr, availabilitySetId, vmssId, osDiskType, vmSize, resKind
'@

$argParams = @{ Query = $kqlVm; First = 1000 }
if ($SubscriptionId) { $argParams.Subscription = $SubscriptionId }

Write-Host 'Inventorying VMs...' -ForegroundColor Cyan
$vms  = Search-AzGraph @argParams
$argParams.Query = $kqlVmss
Write-Host 'Inventorying VMSS...' -ForegroundColor Cyan
$vmss = Search-AzGraph @argParams

$inventory = @($vms) + @($vmss)
Write-Host ("Found {0} compute resources." -f $inventory.Count) -ForegroundColor Green
#endregion

#region SLA tier resolution ---------------------------------------------------
function Resolve-SlaTier {
    param($r)

    $zoneCount = 0
    if ($r.zonesStr) {
        $zoneCount = (@($r.zonesStr -split '[,\s\[\]"]') | Where-Object { $_ -match '^\d+$' }).Count
    }

    if ($r.resKind -eq 'VMSS' -and $zoneCount -ge 2)         { return [pscustomobject]@{ Sla=99.99; Tier='Zones (VMSS)' } }
    if ($r.resKind -eq 'VMSS')                               { return [pscustomobject]@{ Sla=99.95; Tier='VMSS (single zone/region)' } }
    if ($zoneCount -ge 1)                                    { return [pscustomobject]@{ Sla=99.99; Tier='Availability Zone' } }
    if ($r.availabilitySetId)                                { return [pscustomobject]@{ Sla=99.95; Tier='Availability Set' } }

    switch -Regex ($r.osDiskType) {
        'Premium|Ultra'           { return [pscustomobject]@{ Sla=99.9;  Tier='Single VM - Premium/Ultra' } }
        'StandardSSD'             { return [pscustomobject]@{ Sla=99.5;  Tier='Single VM - Standard SSD' } }
        default                   { return [pscustomobject]@{ Sla=95.0;  Tier='Single VM - Standard HDD' } }
    }
}
#endregion

#region Resource Health query -------------------------------------------------
# We pull historical AvailabilityStatuses via REST since the cmdlets are limited.
$token = (Get-AzAccessToken).Token
$headers = @{ Authorization = "Bearer $token"; 'Content-Type' = 'application/json' }

function Get-AvailabilityHistory {
    param([string]$ResourceId)

    $uri = "https://management.azure.com$ResourceId/providers/Microsoft.ResourceHealth/availabilityStatuses?api-version=2022-10-01&`$expand=recommendedactions&`$filter=" +
           [System.Web.HttpUtility]::UrlEncode("properties/availabilityState ne 'NotAvailable' or properties/occuredTime ge $($reportStart.ToString('o'))")
    # Simpler: use list (history) endpoint
    $historyUri = "https://management.azure.com$ResourceId/providers/Microsoft.ResourceHealth/availabilityStatuses?api-version=2022-10-01"
    try {
        $resp = Invoke-RestMethod -Method Get -Uri $historyUri -Headers $headers
        return $resp.value
    } catch {
        Write-Verbose "Health query failed for $ResourceId : $_"
        return @()
    }
}

function Measure-Availability {
    param(
        $events,
        [datetime]$WindowStart = $reportStart,
        [datetime]$WindowEnd   = $reportEnd
    )

    $windowSeconds = ($WindowEnd - $WindowStart).TotalSeconds
    if ($windowSeconds -le 0) {
        return [pscustomobject]@{ AvailablePct = 100.0; UnavailableSeconds = 0; SampleCount = 0; WindowSeconds = 0 }
    }

    if (-not $events -or $events.Count -eq 0) {
        return [pscustomobject]@{ AvailablePct = 100.0; UnavailableSeconds = 0; SampleCount = 0; WindowSeconds = $windowSeconds }
    }

    $sorted = $events |
        ForEach-Object {
            [pscustomobject]@{
                State = $_.properties.availabilityState
                Time  = [datetime]$_.properties.occuredTime
            }
        } | Sort-Object Time

    $unavailableSeconds = 0.0
    for ($i = 0; $i -lt $sorted.Count; $i++) {
        $cur  = $sorted[$i]
        $next = if ($i -lt $sorted.Count - 1) { $sorted[$i+1].Time } else { $WindowEnd }

        $segStart = [Math]::Max($cur.Time.Ticks, $WindowStart.Ticks)
        $segEnd   = [Math]::Min($next.Ticks,    $WindowEnd.Ticks)
        if ($segEnd -le $segStart) { continue }

        if ($cur.State -ne 'Available') {
            $unavailableSeconds += (New-TimeSpan -Start ([datetime]$segStart) -End ([datetime]$segEnd)).TotalSeconds
        }
    }

    $pct = [math]::Round( ((($windowSeconds - $unavailableSeconds) / $windowSeconds) * 100), 4)
    return [pscustomobject]@{
        AvailablePct       = $pct
        UnavailableSeconds = $unavailableSeconds
        SampleCount        = $sorted.Count
        WindowSeconds      = $windowSeconds
    }
}
#endregion

#region Build report ----------------------------------------------------------
$results    = New-Object System.Collections.Generic.List[object]
$eventCache = @{}   # resourceId -> events (reused for region/month matrix)
$counter = 0
foreach ($r in $inventory) {
    $counter++
    Write-Progress -Activity 'Querying Resource Health' -Status $r.name -PercentComplete (($counter / [Math]::Max($inventory.Count,1)) * 100)

    $sla     = Resolve-SlaTier $r
    $events  = Get-AvailabilityHistory -ResourceId $r.id
    $eventCache[$r.id] = $events
    $measure = Measure-Availability -events $events -WindowStart $reportStart -WindowEnd $reportEnd

    $met = $measure.AvailablePct -ge $sla.Sla

    $results.Add([pscustomobject]@{
        Subscription       = $r.subscriptionId
        ResourceGroup      = $r.resourceGroup
        Name               = $r.name
        Kind               = $r.resKind
        Location           = $r.location
        VmSize             = $r.vmSize
        OsDiskType         = $r.osDiskType
        SlaTier            = $sla.Tier
        CommittedSlaPct    = $sla.Sla
        MeasuredAvailPct   = $measure.AvailablePct
        UnavailableMinutes = [math]::Round($measure.UnavailableSeconds / 60, 2)
        SlaMet             = $met
        ResourceId         = $r.id
    })
}
Write-Progress -Activity 'Querying Resource Health' -Completed
#endregion

#region Region x Month availability matrix ------------------------------------
# Build list of months (oldest -> newest) ending with $Month
$monthList = @()
$anchor = [datetime]::ParseExact("$Month-01", 'yyyy-MM-dd', $null)
for ($i = $MatrixMonths - 1; $i -ge 0; $i--) {
    $monthList += $anchor.AddMonths(-$i)
}

# Aggregate: per region+month accumulate downtime seconds and total seconds across resources
$matrix = @{}   # key "region|yyyy-MM" -> @{ Down=...; Total=... }
foreach ($r in $inventory) {
    $events = $eventCache[$r.id]
    foreach ($m in $monthList) {
        $ws = $m.ToUniversalTime()
        $we = $ws.AddMonths(1)
        if ($we -gt $now) { $we = $now }
        if ($we -le $ws) { continue }

        $mr = Measure-Availability -events $events -WindowStart $ws -WindowEnd $we
        $key = '{0}|{1}' -f $r.location, $m.ToString('yyyy-MM')
        if (-not $matrix.ContainsKey($key)) { $matrix[$key] = @{ Down = 0.0; Total = 0.0 } }
        $matrix[$key].Down  += $mr.UnavailableSeconds
        $matrix[$key].Total += $mr.WindowSeconds
    }
}

$regions = $inventory | Select-Object -ExpandProperty location -Unique | Sort-Object
$monthKeys = $monthList | ForEach-Object { $_.ToString('yyyy-MM') }

# Build CSV-friendly rows
$matrixRows = foreach ($region in $regions) {
    $row = [ordered]@{ Region = $region }
    foreach ($mk in $monthKeys) {
        $key = "$region|$mk"
        if ($matrix.ContainsKey($key) -and $matrix[$key].Total -gt 0) {
            $pct = (($matrix[$key].Total - $matrix[$key].Down) / $matrix[$key].Total) * 100
            $row[$mk] = [math]::Round($pct, 4)
        } else {
            $row[$mk] = $null
        }
    }
    [pscustomobject]$row
}
#endregion

#region Output ---------------------------------------------------------------
$stamp   = Get-Date -Format 'yyyyMMdd-HHmmss'
$csvPath    = Join-Path $OutputFolder "AzComputeSla_$Month`_$stamp.csv"
$matrixCsv  = Join-Path $OutputFolder "AzComputeSla_RegionMatrix_$Month`_$stamp.csv"
$htmPath    = Join-Path $OutputFolder "AzComputeSla_$Month`_$stamp.html"

$results | Sort-Object SlaMet, MeasuredAvailPct |
    Export-Csv -Path $csvPath -NoTypeInformation -Encoding UTF8

$matrixRows | Export-Csv -Path $matrixCsv -NoTypeInformation -Encoding UTF8

$breaches = $results | Where-Object { -not $_.SlaMet }
$summary  = [pscustomobject]@{
    Month            = $Month
    Generated        = (Get-Date).ToString('u')
    TotalResources   = $results.Count
    SlaBreaches      = $breaches.Count
    OverallAvgAvail  = if ($results.Count) { [math]::Round((($results | Measure-Object MeasuredAvailPct -Average).Average),4) } else { 0 }
}

$style = @'
<style>
body{font-family:Segoe UI,Arial;margin:24px;color:#222}
h1{margin:0 0 4px 0}h2{margin-top:32px}
table{border-collapse:collapse;width:100%;font-size:13px;margin-top:8px}
th,td{border:1px solid #ddd;padding:6px 8px;text-align:left}
th{background:#0a3a6b;color:#fff;text-align:center}
tr.bad{background:#fde7e9}
tr.good{background:#eaf7ea}
.kpi{display:inline-block;margin-right:24px;padding:12px 18px;border-radius:8px;background:#f3f3f3}
.kpi b{font-size:22px;display:block}
.matrix td{text-align:center}
.matrix td.region{text-align:left;font-weight:600;background:#fafafa}
.matrix td.warn{background:#fff4ce}
.matrix td.bad {background:#fde7e9}
.matrix td.na  {color:#888;font-style:italic}
.note{font-size:12px;color:#555;margin-top:6px}
</style>
'@

$kpiHtml = @"
<div class="kpi"><b>$($summary.TotalResources)</b>Compute resources</div>
<div class="kpi"><b>$($summary.OverallAvgAvail)%</b>Avg availability</div>
<div class="kpi"><b>$($summary.SlaBreaches)</b>SLA breaches</div>
"@

$rowsHtml = ($results | Sort-Object SlaMet, MeasuredAvailPct | ForEach-Object {
    $cls = if ($_.SlaMet) { 'good' } else { 'bad' }
    "<tr class='$cls'><td>$($_.Subscription)</td><td>$($_.ResourceGroup)</td><td>$($_.Name)</td><td>$($_.Kind)</td><td>$($_.Location)</td><td>$($_.VmSize)</td><td>$($_.SlaTier)</td><td>$($_.CommittedSlaPct)%</td><td>$($_.MeasuredAvailPct)%</td><td>$($_.UnavailableMinutes)</td><td>$($_.SlaMet)</td></tr>"
}) -join "`n"

# Region x Month matrix HTML
$matrixHeader = "<tr><th>Region</th>" + (($monthKeys | ForEach-Object {
    "<th>$([datetime]::ParseExact("$_-01",'yyyy-MM-dd',$null).ToString('MMM-yy'))<br/>COMPUTE</th>"
}) -join '') + "</tr>"

$matrixBody = ($matrixRows | ForEach-Object {
    $row = $_
    $cells = foreach ($mk in $monthKeys) {
        $v = $row.$mk
        if ($null -eq $v) {
            "<td class='na'>n/a</td>"
        } else {
            $cls = if     ($v -lt 99.9)  { 'bad'  }
                   elseif ($v -lt 99.95) { 'warn' }
                   else                  { ''     }
            "<td class='$cls'>$([string]::Format('{0:F4}%', $v))</td>"
        }
    }
    "<tr><td class='region'>$($row.Region)</td>$([string]::Join('', $cells))</tr>"
}) -join "`n"

$html = @"
<!doctype html><html><head><meta charset='utf-8'><title>Azure Compute SLA Report - $Month</title>$style</head><body>
<h1>Azure Compute Availability SLA Report</h1>
<div>Reporting month: <b>$Month</b> &nbsp;|&nbsp; Generated: $($summary.Generated)</div>
<div style='margin-top:16px'>$kpiHtml</div>

<h2>Cumulative Compute Uptime &mdash; per Region per Month (last $MatrixMonths months)</h2>
<table class='matrix'>
<thead>$matrixHeader</thead>
<tbody>
$matrixBody
</tbody></table>
<div class='note'>Aggregated across all VMs/VMSS in each region. Yellow &lt; 99.95%, red &lt; 99.9%. Months with no Resource Health events are reported as 100% (Microsoft Resource Health retains ~30 days of events; for full historical fidelity export ResourceHealth events to Log Analytics).</div>

<h2>Resources (current month: $Month)</h2>
<table>
<thead><tr><th>Subscription</th><th>Resource Group</th><th>Name</th><th>Kind</th><th>Location</th><th>Size</th><th>SLA Tier</th><th>Committed</th><th>Measured</th><th>Unavail (min)</th><th>SLA Met</th></tr></thead>
<tbody>
$rowsHtml
</tbody></table>
</body></html>
"@

$html | Out-File -FilePath $htmPath -Encoding UTF8

Write-Host ''
Write-Host "CSV    : $csvPath"   -ForegroundColor Green
Write-Host "Matrix : $matrixCsv" -ForegroundColor Green
Write-Host "HTML   : $htmPath"   -ForegroundColor Green
Write-Host ("Breaches: {0} / {1}" -f $summary.SlaBreaches, $summary.TotalResources) -ForegroundColor $(if($summary.SlaBreaches){'Yellow'}else{'Green'})
#endregion
